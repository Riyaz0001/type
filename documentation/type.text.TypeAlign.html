Class: TypeAlign

TypeAlign

new type.text.TypeAlign()

textfield/align/customModule/TypeAlign.js, line 47
Used to Define a custom aligment for the chars in the textfield
Example
var typeAlign = new type.text.TypeAlign();
typeAlign.startingX = 50;
typeAlign.startingY = 250;
typeAlign.steps = [{
    "x": 20,
    "y": 70,
    "rotation": 90
}, {
    "x": 15,
    "y": -20
}, {
    "x": 15,
    "y": +20,
    "rotation": 45
}, {
    "x": 15,
    "y": -70
},
{
    "x": 20,
    "y": -70,
    "rotation": 45
}, {
    "x": 15,
    "y": +20
}, {
    "x": 15,
    "y": -20
}, {
    "x": 15,
    "y": +70
}];


txField.typeAlign = typeAlign;

txField.customAlign = true;
